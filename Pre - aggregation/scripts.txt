# Start Kafka
bin/kafka-server-start.sh -daemon config/server.properties

# Create kafka topic
bin/kafka-topics.sh --zookeeper localhost:2191 --create --topic user_stream_startree --partitions 1 --replication-factor 1

# Product data to kafka topic
cat /home/centos/pinot/kafka_2.12-2.4.0/user_data_stream_startree.txt | bin/kafka-console-producer.sh --broker-list localhost:9092 --topic user_stream_startree 

# Run query to group by multiple values
select sum(views) as viewCount,country,device,subscription_plan from user_stream_startree group by country, device, subscription_plan order by viewCount desc
